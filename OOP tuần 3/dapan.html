Lớp Point — 2D Geometry
Class · Method ▾

Tạo lớp Point với thuộc tính x, y kiểu int. Thực hiện:

  • Tạo điểm A(3, 4) → hiển thị tọa độ
  • Tạo điểm B từ bàn phím
  • Tạo điểm C đối xứng với B qua gốc O
  • Tính khoảng cách từ B đến O: d = √(x² + y²)
  • Tính khoảng cách từ A đến B: d = √((x₁−x₂)² + (y₁−y₂)²)
point.py 
import math

class Point:
    def __init__(self, x: int, y: int):
        self.x = x
        self.y = y

    def hien_thi(self):
        print(f"Điểm ({self.x}, {self.y})")

    def doi_xung_qua_O(self):
        # Điểm đối xứng qua gốc O: (-x, -y)
        return Point(-self.x, -self.y)

    def khoang_cach_den_O(self):
        return math.sqrt(self.x**2 + self.y**2)

    def khoang_cach_den(self, other):
        return math.sqrt((self.x - other.x)**2 + (self.y - other.y)**2)


# ---- Tạo điểm A(3, 4) ----
A = Point(3, 4)
print("Điểm A:"); A.hien_thi()

# ---- Nhập điểm B từ bàn phím ----
xb = int(input("Nhập x của B: "))
yb = int(input("Nhập y của B: "))
B = Point(xb, yb)
print("Điểm B:"); B.hien_thi()

# ---- Điểm C đối xứng với B qua O ----
C = B.doi_xung_qua_O()
print("Điểm C (đối xứng B qua O):"); C.hien_thi()

# ---- Khoảng cách B → O ----
print(f"Khoảng cách B đến O: {B.khoang_cach_den_O():.2f}")

# ---- Khoảng cách A → B ----
print(f"Khoảng cách A đến B: {A.khoang_cach_den(B):.2f}")

Điểm B: (1, 2)
Điểm C (đối xứng B qua O): (-1, -2)
Khoảng cách B đến O: 2.24
Khoảng cách A đến B: 2.83

Tạo lớp SieuNhan với thuộc tính: ten, vu_khi, mau_sac. Khởi tạo 2 đối tượng và in ra thể hiện của cả hai.

While · For · List ▾

Chương trình quản lý siêu nhân toàn thế giới:

  • Khởi tạo class siêu nhân (thuộc tính tùy chọn)
  • Dùng while để nhập danh sách
  • Dùng for để in danh sách kèm thuộc tính

class SieuNhan:

        print(f"{self.ten} đang ăn {thuc_an}. Ngon!")

    def chay(self, toc_do):
        print(f"{self.ten} chạy với tốc độ {toc_do} km/h!")

class OTo:
    def __init__(self, hang, kich_thuoc, mau, gia):
        self.hang       = hang
        self.kich_thuoc = kich_thuoc
        self.mau        = mau
        self.gia        = gia
        self.toc_do     = 0

    def tang_toc(self, them):
        self.toc_do += them
        print(f"Tăng tốc! Tốc độ hiện tại: {self.toc_do} km/h")

    def giam_toc(self, bot):
        self.toc_do = max(0, self.toc_do - bot)
        print(f"Giảm tốc! Tốc độ hiện tại: {self.toc_do} km/h")

    def dam(self):
        print(f"💥 Xe {self.hang} bị đâm! Dừng khẩn cấp.")
        self.toc_do = 0

class TaiKhoan:
    def __init__(self, ten_tk, so_tk, ngan_hang, so_du=0):
        self.ten_tk    = ten_tk
        self.so_tk     = so_tk
        self.ngan_hang = ngan_hang
        self.__so_du   = so_du  # thuộc tính private

    def rut(self, so_tien):
        if so_tien > self.__so_du:
            print("❌ Số dư không đủ!")
        else:
            self.__so_du -= so_tien
            print(f"✅ Rút {so_tien:,} VNĐ. Số dư còn: {self.__so_du:,}")

    def gui(self, so_tien):
        self.__so_du += so_tien
        print(f"✅ Gửi {so_tien:,} VNĐ. Số dư mới: {self.__so_du:,}")

    def kiem_tra_so_du(self):
        print(f"💳 [{self.ten_tk}] Số dư: {self.__so_du:,} VNĐ")

Think Python — Circle Class
Chapter 15 · Exercise 15.1 ▾

Xây dựng lớp Circle với center (Point) và radius (float). Viết các hàm:

  • point_in_circle: kiểm tra điểm có nằm trong/trên vòng tròn không
  • rect_in_circle: kiểm tra hình chữ nhật có nằm hoàn toàn trong vòng tròn không
  • rect_circle_overlap: kiểm tra hình chữ nhật có giao với vòng tròn không
circle.py 
import math

class Point:
    def __init__(self, x=0.0, y=0.0):
        self.x = x
        self.y = y

class Rectangle:
    def __init__(self, corner: Point, width: float, height: float):
        self.corner = corner   # góc dưới-trái
        self.width  = width
        self.height = height

class Circle:
    def __init__(self, center: Point, radius: float):
        self.center = center
        self.radius = radius


def distance(p1: Point, p2: Point) -> float:
    return math.sqrt((p1.x - p2.x)**2 + (p1.y - p2.y)**2)


def point_in_circle(circle: Circle, point: Point) -> bool:
    """True nếu điểm nằm trong hoặc trên vòng tròn."""
    return distance(circle.center, point) <= circle.radius


def rect_in_circle(circle: Circle, rect: Rectangle) -> bool:
    """True nếu HCN nằm hoàn toàn trong/trên vòng tròn."""
    c = rect.corner
    corners = [
        Point(c.x,              c.y),
        Point(c.x + rect.width, c.y),
        Point(c.x,              c.y + rect.height),
        Point(c.x + rect.width, c.y + rect.height),
    ]
    return all(point_in_circle(circle, p) for p in corners)


def rect_circle_overlap(circle: Circle, rect: Rectangle) -> bool:
    """True nếu bất kỳ góc nào của HCN nằm trong vòng tròn."""
    c = rect.corner
    corners = [
        Point(c.x,              c.y),
        Point(c.x + rect.width, c.y),
        Point(c.x,              c.y + rect.height),
        Point(c.x + rect.width, c.y + rect.height),
    ]
    return any(point_in_circle(circle, p) for p in corners)


# ---- Demo ----
circle = Circle(Point(150, 100), 75)

p1 = Point(150, 100)   # tâm → trong
p2 = Point(300, 300)   # xa → ngoài
print(f"p1 trong vòng tròn: {point_in_circle(circle, p1)}")
print(f"p2 trong vòng tròn: {point_in_circle(circle, p2)}")

rect_small = Rectangle(Point(140, 90), 10, 10)
rect_large = Rectangle(Point(100, 50), 300, 200)
print(f"rect_small trong circle: {rect_in_circle(circle, rect_small)}")
print(f"rect_large overlap circle: {rect_circle_overlap(circle, rect_large)}")

p2 trong vòng tròn: False
rect_small trong circle: True

Bài Tập Nâng Cao — Lớp NhanVien
Advanced · Getter/Setter ▾

Xây dựng lớp NhanVien với đầy đủ getter/setter và các nghiệp vụ:

  • tinhLuong(): Lương = luongCoban × heSoLuong
  • inTTin(): Hiển thị đầy đủ thông tin nhân viên
  • tangLuong(delta): Nếu lương mới > LUONG_MAX → in cảnh báo, trả False; ngược lại trả True
  • Getter/setter cho tất cả thuộc tính
nhanvien.py 
class NhanVien:
    LUONG_MAX = 50_000_000  # Class variable (static)

    def __init__(self, ten_nhan_vien: str,
                 luong_co_ban: float, he_so_luong: float):
        self.__tenNhanVien  = ten_nhan_vien
        self.__luongCoBan   = luong_co_ban
        self.__heSoLuong    = he_so_luong

    # ── Getter / Setter ─────────────────────────────────
    @property
    def tenNhanVien(self): return self.__tenNhanVien

    @tenNhanVien.setter
    def tenNhanVien(self, value):
        if not value.strip():
            raise ValueError("Tên không được rỗng!")
        self.__tenNhanVien = value

    @property
    def luongCoBan(self): return self.__luongCoBan

    @luongCoBan.setter
    def luongCoBan(self, value):
        if value < 0:
            raise ValueError("Lương không được âm!")
        self.__luongCoBan = value

    @property
    def heSoLuong(self): return self.__heSoLuong

    @heSoLuong.setter
    def heSoLuong(self, value):
        if value <= 0:
            raise ValueError("Hệ số lương phải > 0!")
        self.__heSoLuong = value

    # ── Phương thức nghiệp vụ ────────────────────────────
    def tinhLuong(self) -> float:
        return self.__luongCoBan * self.__heSoLuong

    def inTTin(self):
        luong = self.tinhLuong()
        print(f"""
╔══════════════════════════════════════╗
║        THÔNG TIN NHÂN VIÊN           ║
╠══════════════════════════════════════╣
║ Tên       : {self.__tenNhanVien:<25}║
║ Lương CB  : {self.__luongCoBan:>20,.0f} VNĐ ║
║ Hệ số     : {self.__heSoLuong:>25.1f} ║
║ Lương TT  : {luong:>20,.0f} VNĐ ║
╚══════════════════════════════════════╝""")

    def tangLuong(self, delta: float) -> bool:
        luong_moi = (self.__luongCoBan + delta) * self.__heSoLuong
        if luong_moi > NhanVien.LUONG_MAX:
            print(f"⚠️  Lương mới ({luong_moi:,.0f}) vượt LUONG_MAX. Không tăng!")
            return False
        self.__luongCoBan += delta
        print(f"✅ Đã tăng lương! Lương mới: {self.tinhLuong():,.0f} VNĐ")
        return True


# ---- Demo ----
nv = NhanVien("Nguyễn Văn A", 10_000_000, 3.0)
nv.inTTin()
nv.tangLuong(5_000_000)   # → 45M < 50M: OK
nv.tangLuong(10_000_000)  # → 75M > 50M: Cảnh báo
